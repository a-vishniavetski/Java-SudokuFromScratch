6   public class SudokuBoard {
7   
8       //Deklaracja zmiennej board o wymiarach 9 x 9
9       int[][] board = new int[9][9];
10  
11      //konstruktor klasy
12      public SudokuBoard() {
13  
14      }
15  
16      public void solveGame() {
17          BacktrackingSudokuSolver sudokuSolver = new BacktrackingSudokuSolver();
18          sudokuSolver.solve(this);
19      }
20  
21      //Funkcja wypisująca całą tablicę sudoku w formacie 9 x 9
22      public void printBoard() {
23          for (int x = 0; x < 9; x++) {
24              for (int y = 0; y < 9; y++) {
25                  System.out.print(" " + get(x, y) + " ");
26              }
27              System.out.print("\n");
28          }
29      }
30  
31      //Funkcja zwracająca kopię tablicy sudoku, ale w formie, w której niemożliwe jest
32      //modyfikowanie jej z zewnątrz
33      public int[][] getBoard() {
34          int[][] tab = new int[9][9];
35          for (int x = 0; x < 9; x++) {
36              System.arraycopy(board[x], 0, tab[x], 0, 9);
37          }
38          return tab;
39      }
40  
41      //Funkcja zwracająca liczbę z danych koordynatów x y
42      public int get(int x, int y) {
43          return board[x][y];
44      }
45  
46      //Funkcja ustawiająca liczbę na danych koordynatach
47      void set(int x, int y, int num) {
48          board[x][y] = num;
49      }
50  
51  
52  
53      //Funkcja sprawdzająca całą tablicę, czy jest ona poprawna wg zasad sudoku
54      boolean checkBoard(int num, int x, int y) {
55          if (!checkRow(num, x)) {
56              return false;
57          }
58          if (!checkColumn(num, y)) {
59              return false;
60          }
61          if (!check3by3(num, x, y)) {
62              return false;
63          }
64          return true;
65      }
66  
67      //Funkcja generująca losową cyfrę od 1 do 9
68  
69      //Funkcja sprawdzająca danych rząd wg zasad sudoku
70      private boolean checkRow(int num, int x) {
71          for (int y = 0; y < 9; y++) {
72              if (get(x, y) == num) {
73                  return false;
74              }
75          }
76          return true;
77      }
78  
79      //FUnkcja sprawdzająca daną kolumnę wg zasad sudoku
80      private boolean checkColumn(int num, int y) {
81          for (int x = 0; x < 9; x++) {
82              if (get(x, y) == num) {
83                  return false;
84              }
85          }
86          return true;
87      }
88  
89      //Funkcja sprawdzająca dany kwadrat wg zasad sudoku
90      private boolean check3by3(int num, int x, int y) {
91          x = getXOfSquare(x);
92          y = getYOfSquare(y);
93          for (int i = x; i < x + 3; i++) {
94              for (int j = y; j < y + 3; j++) {
95                  if (get(i, j) == num) {
96                      return false;
97                  }
98              }
99          }
100         return true;
101     }
102 
103     //Funkcja znajdująca pierwszy rząd (lewy górny róg) kwadratu po podaniu danej x
104     private int getXOfSquare(int x) {
105         return x - (x % 3);
106     }
107 
108     //Funkcja znajdująca pierwszą kolumnę (lewy górny róg) kwadratu po podaniu danej y
109     private int getYOfSquare(int y) {
110         return y - (y % 3);
111     }
112 }
